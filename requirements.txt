# ── GitHub credentials ─────────────────────────────────────────────────────────
#
# Use a fine-grained Personal Access Token (not a classic token).
# How to create one:
#   GitHub → Settings → Developer settings → Personal access tokens → Fine-grained tokens
#
# Scope it ONLY to the repositories this service will patch.
# Required permissions (nothing else):
#   Repository permissions → Contents:       Read and write
#   Repository permissions → Pull requests:  Read and write
#
# A classic token with the broad 'repo' scope gives write access to every private
# repo you own — that's the minimum-privilege violation the reviewer flagged.
#
GITHUB_TOKEN=github_pat_your_fine_grained_token_here


# ── Groq credentials ───────────────────────────────────────────────────────────

GROQ_API_KEY=gsk_your_groq_api_key_here


# ── Webhook secrets ────────────────────────────────────────────────────────────
#
# WHY per-repo secrets?
# If you use a single shared secret and it leaks, an attacker can forge valid
# webhook payloads for every repo this service handles. Per-repo secrets limit
# the blast radius to just that one repo.
#
# Naming convention: WEBHOOK_SECRET_{OWNER}_{REPO}
#   (uppercase, hyphens and dots → underscores)
#
# Examples:
#   github.com/alice/my-api       →  WEBHOOK_SECRET_ALICE_MY_API
#   github.com/Sarvjais12/test-auto-patcher  →  WEBHOOK_SECRET_SARVJAIS12_TEST_AUTO_PATCHER
#
WEBHOOK_SECRET_SARVJAIS12_TEST_AUTO_PATCHER=a_long_random_string_for_this_specific_repo

# Global fallback — used only if no per-repo secret is found.
# Migrate each repo to a dedicated secret when you can.
GITHUB_WEBHOOK_SECRET=a_different_long_random_string_for_the_fallback
